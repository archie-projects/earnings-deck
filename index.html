Modeling EPS (Python)


import pandas as pd

# Hypothetical shares & net income
net_income = pd.Series([780, 860, 920], index=["Q4", "Q1", "Q2"])  # $M
shares_out = 640  # million shares

# EPS calculation
EPS = (net_income * 1e6) / (shares_out * 1e6)
EPS = EPS.round(2)
print(EPS)

# Simple growth vs previous quarter
qoq_growth = (EPS["Q2"] - EPS["Q1"]) / EPS["Q1"]
print(f"QoQ EPS growth: {qoq_growth:.1%}")
